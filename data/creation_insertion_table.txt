use thermogreen;

-- Supprimer la table mesure si elle existe
DROP TABLE IF EXISTS mesure;

-- Créer la table mesure avec auto-incrément
CREATE TABLE mesure (
    id_mesure INT AUTO_INCREMENT PRIMARY KEY,
    numero VARCHAR(2),
    date_mesure DATETIME,
    fahrenheit DECIMAL(5, 1)
);

-- Supprimer la table user si elle existe
DROP TABLE IF EXISTS user;

-- Créer la table user avec login et password
CREATE TABLE user (
    login VARCHAR(255) PRIMARY KEY,
    password VARCHAR(255)
);
